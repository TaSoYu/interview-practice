## 思路：
### 1.暴力做法： 用一个额外数组 存储数据，再铺回去



## Code
### 1.
```
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> a(n);

        for(int i=0; i<nums.size(); i++)
        {
            int idx = (i + k) % n;
            a[idx] = nums[i]; 
        }
        for(int i=0; i<n; i++)
        {
            nums[i] = a[i];
        }

    }
};
```
